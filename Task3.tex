	\begin{subtask}{6}{Complete the following sequence diagram.}{1.5}{}{}{}{}
		\begin{problem}
			\begin{sequencediagram}
				\def\unitfactor{0.9}
				\newthread{useragent}{:\ User Agent}
				\newinst[7.5cm]{server1}{ex.org :\ Server}
				\begin{call}{useragent}{ \foreach \n in {1,...,5}{\_} /mygeo HTTP/1.1 }
				{server1}{\shortstack{HTTP/1.1 \foreach \n in {1,...,5}{\_} Not found
				\\ \foreach \n in {1,...,15}{\_}: text/html; charset=utf-8
				}}
				\postlevel
				\end{call}
			\end{sequencediagram}
		\end{problem}
	
		\begin{solution}
			\textbf{Solution:}
			\begin{sequencediagram}
				\def\unitfactor{0.9}
				\newthread{useragent}{:\ User Agent}
				\newinst[7.5cm]{server1}{ex.org :\ Server}
				\begin{call}{useragent}{ GET /mygeo HTTP/1.1 }
				{server1}{\shortstack{HTTP/1.1 404 Not found
				\\ Content-Type: text/html; charset=utf-8
				}}
				\postlevel
				\end{call}
			\end{sequencediagram}
		\end{solution}
	\end{subtask}
